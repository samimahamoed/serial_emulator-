152 void SettingsDialog::fillPortsInfo()
153 {
154  ui->serialPortInfoListBox->clear();
155  foreach (const QSerialPortInfo &info, QSerialPortInfo::availablePorts()) {
156  QStringList list;
157  list << info.portName()
158  << info.description()
159  << info.manufacturer()
160  << info.systemLocation()
161  << (info.vendorIdentifier() ? QString::number(info.vendorIdentifier(), 16) : QString())
162  << (info.productIdentifier() ? QString::number(info.productIdentifier(), 16) : QString());
163 
164  ui->serialPortInfoListBox->addItem(list.first(), list);
165  }
166 }

168 void SettingsDialog::updateSettings()
169 {
170  currentSettings.name = ui->serialPortInfoListBox->currentText();
171 
172  // Baud Rate
173  if (ui->baudRateBox->currentIndex() == 4) {
174  // custom baud rate
175  currentSettings.baudRate = ui->baudRateBox->currentText().toInt();
176  } else {
177  // standard baud rate
178  currentSettings.baudRate = static_cast<QSerialPort::BaudRate>(
179  ui->baudRateBox->itemData(ui->baudRateBox->currentIndex()).toInt());
180  }
181  currentSettings.stringBaudRate = QString::number(currentSettings.baudRate);
182 
183  // Data bits
184  currentSettings.dataBits = static_cast<QSerialPort::DataBits>(
185  ui->dataBitsBox->itemData(ui->dataBitsBox->currentIndex()).toInt());
186  currentSettings.stringDataBits = ui->dataBitsBox->currentText();
187 
188  // Parity
189  currentSettings.parity = static_cast<QSerialPort::Parity>(
190  ui->parityBox->itemData(ui->parityBox->currentIndex()).toInt());
191  currentSettings.stringParity = ui->parityBox->currentText();
192 
193  // Stop bits
194  currentSettings.stopBits = static_cast<QSerialPort::StopBits>(
195  ui->stopBitsBox->itemData(ui->stopBitsBox->currentIndex()).toInt());
196  currentSettings.stringStopBits = ui->stopBitsBox->currentText();
197 
198  // Flow control
199  currentSettings.flowControl = static_cast<QSerialPort::FlowControl>(
200  ui->flowControlBox->itemData(ui->flowControlBox->currentIndex()).toInt());
201  currentSettings.stringFlowControl = ui->flowControlBox->currentText();
202 
203  // Additional options
204  currentSettings.localEchoEnabled = ui->localEchoCheckBox->isChecked();
205 }
